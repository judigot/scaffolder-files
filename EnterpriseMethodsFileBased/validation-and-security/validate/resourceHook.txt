import { useMutation, type UseMutationOptions } from '@tanstack/react-query';
import { validateAndSanitize } from '@/hooks/shared/useValidate.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IValidationData {
  data: Partial<I{{tableName.singular.pascalCase}}>;
  rules?: Record<string, unknown>;
}

export const useValidate{{tableName.singular.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<I{{tableName.singular.pascalCase}}, Error, IValidationData>,
    'mutationFn'
  >,
) => {
  return useMutation({
    mutationFn: ({ data, rules }: IValidationData) =>
      validateAndSanitize<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}',
        data,
        rules
      ),
    ...options,
  });
};
