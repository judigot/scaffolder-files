import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { useDestroy } from '@/hooks/shared/useDestroy.ts';

export const useDelete{{tableName.singular.pascalCase}} = (
  options?: Omit<UseMutationOptions<void, Error, number | string>, 'mutationFn'>,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: useDestroy,
    onSuccess: (_, id) => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
      queryClient.removeQueries({ queryKey: ['{{tableName.singular.camelCase}}', id] });
    },
    ...options,
  });
};
