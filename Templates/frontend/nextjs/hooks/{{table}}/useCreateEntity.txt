import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { createResource } from '@/hooks/shared/useCreateResource.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

export const useCreate{{tableName.singular.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<I{{tableName.singular.pascalCase}}, Error, Omit<I{{tableName.singular.pascalCase}}, 'id'>>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: createResource,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
    },
    ...options,
  });
};
