import express from 'express';
import { router } from './routes/index.js';

const app = express();
const PORT = process.env.PORT || 3001;

app.use(express.json());
app.use('/api', router);

app.get('/', (_req, res) => {
  res.json({ message: 'Express API', docs: '/api' });
});

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
