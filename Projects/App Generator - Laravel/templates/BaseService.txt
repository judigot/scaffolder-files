<?php

namespace App\Services;

use App\Repositories\BaseRepository;
use Illuminate\Support\Collection;
use Illuminate\Database\Eloquent\Model;

abstract class BaseService
{
    protected BaseRepository $repository;

    public function __construct(BaseRepository $repository)
    {
        $this->repository = $repository;
    }

    public function getSortedQuery(?string $column = null, string $direction = 'asc'): \Illuminate\Database\Eloquent\Builder 
    {
        return $this->repository->getSortedQuery ($column, $direction); 
    }

    public function create(array $data): Model
    {
        return $this->repository->create($data);
    }

    public function index(?string $column = null, string $direction = 'asc'): Collection
    {
        return $this->repository->index($column, $direction);
    }

    public function findById(string $id): ?Model
    {
        return $this->repository->findById($id);
    }

    public function update(string $id, array $data): bool
    {
        return $this->repository->update($id, $data);
    }

    public function destroy(string $id): bool
    {
        return $this->repository->destroy($id);
    }
}
