import { NextRequest, NextResponse } from 'next/server';
import DatatypeParser from '@/utils/DataTypeParser.ts';
import { prisma } from '@/prisma/DatabaseClient';
import { <@@>tableName</@@> } from '@prisma/client';
import { <@@>tableName.pascalCase</@@> } from '@/Controllers/<@@>tableName.pascalCase</@@>Controller.ts';

type Body = Omit<<@@>tableName</@@>, '<@@>primaryKey</@@>'>

const PatchHandler = async (req: NextRequest) => {
  try {
    const body: Body = await req.json();
    const result = await <@@>tableName.pascalCase</@@>.patch(body);
    return NextResponse.json(DatatypeParser(result));
  } catch (error) {
    console.error(error);
    return NextResponse.json({
      error: error,
    });
  } finally {
    await prisma.$disconnect();
  }
};

export default PatchHandler;
