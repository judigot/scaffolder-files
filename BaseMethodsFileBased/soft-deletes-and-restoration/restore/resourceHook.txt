export const useRestore{{tableNamePascalCaseSingular}} = (
  options?: Omit<
    UseMutationOptions<I{{tableNamePascalCaseSingular}}, Error, string | number>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (id: string | number) =>
      restoreResource<I{{tableNamePascalCaseSingular}}>(
        `/{{tableNameCamelCasePlural}}/${String(id)}/restore`

    onSuccess: () => {

};
