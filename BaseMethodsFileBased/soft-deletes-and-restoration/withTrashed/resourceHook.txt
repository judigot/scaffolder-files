import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { getWithTrashedResources } from '@/hooks/shared/useWithTrashed.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IWithTrashedParams {
  filters?: Record<string, unknown>;
  limit?: number;
  orderBy?: string;
  orderDirection?: 'asc' | 'desc';
}

export const useWithTrashed{{tableName.plural.pascalCase}} = (
  params: IWithTrashedParams = {},
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'withTrashed', params.filters, params.limit, params.orderBy, params.orderDirection],
    queryFn: () =>
      getWithTrashedResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/with-trashed',
        params.filters,
        params.limit,
        params.orderBy,
        params.orderDirection
      ),
    ...options,
  });
};
