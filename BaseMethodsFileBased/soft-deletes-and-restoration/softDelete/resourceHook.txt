import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { softDeleteResource } from '@/hooks/shared/useSoftDelete.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

export const useSoftDelete{{tableName.singular.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<I{{tableName.singular.pascalCase}}, Error, string | number>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (id: string | number) =>
      softDeleteResource<I{{tableName.singular.pascalCase}}>(
        `/{{tableName.plural.camelCase}}/${String(id)}/soft-delete`
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
    },
    ...options,
  });
};
