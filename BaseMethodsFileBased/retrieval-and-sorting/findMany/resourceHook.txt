import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { findManyResources } from '@/hooks/shared/useFindMany.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IFindManyParams {
  ids: (string | number)[];
  orderBy?: string;
  orderDirection?: 'asc' | 'desc';
}

export const useFindMany{{tableName.plural.pascalCase}} = (
  params: IFindManyParams,
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'findMany', params.ids, params.orderBy, params.orderDirection],
    queryFn: () =>
      findManyResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/find-many',
        params.ids,
        params.orderBy,
        params.orderDirection
      ),
    enabled: Boolean(params.ids && params.ids.length > 0),
    ...options,
  });
};
