import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { orderByResources } from '@/hooks/shared/useOrderBy.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IOrderByParams {
  column: string;
  direction?: 'asc' | 'desc';
  limit?: number;
  filters?: Record<string, unknown>;
}

export const useOrderBy{{tableName.plural.pascalCase}} = (
  params: IOrderByParams,
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'orderBy', params.column, params.direction, params.limit, params.filters],
    queryFn: () =>
      orderByResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/order-by',
        params.column,
        params.direction,
        params.limit,
        params.filters
      ),
    enabled: Boolean(params.column),
    ...options,
  });
};
