import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { updateOrCreateResource } from '@/hooks/shared/useUpdateOrCreate.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IUpdateOrCreateData {
  searchCriteria: Partial<I{{tableName.singular.pascalCase}}>;
  data: Omit<I{{tableName.singular.pascalCase}}, 'id'>;
}

export const useUpdateOrCreate{{tableName.singular.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<I{{tableName.singular.pascalCase}}, Error, IUpdateOrCreateData>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ searchCriteria, data }: IUpdateOrCreateData) =>
      updateOrCreateResource<I{{tableName.singular.pascalCase}}, Omit<I{{tableName.singular.pascalCase}}, 'id'>>(
        '/{{tableName.plural.camelCase}}/update-or-create',
        searchCriteria,
        data
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
    },
    ...options,
  });
};
