import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { batchUpdateResources } from '@/hooks/shared/useBatchUpdate.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IBatchUpdateData {
  ids: (number | string)[];
  data: Partial<I{{tableName.singular.pascalCase}}>;
}

export const useBatchUpdate{{tableName.plural.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<I{{tableName.singular.pascalCase}}[], Error, IBatchUpdateData>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ ids, data }: IBatchUpdateData) =>
      batchUpdateResources<I{{tableName.singular.pascalCase}}, Partial<I{{tableName.singular.pascalCase}}>>(
        '/{{tableName.plural.camelCase}}/batch-update',
        ids,
        data
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
    },
    ...options,
  });
};
