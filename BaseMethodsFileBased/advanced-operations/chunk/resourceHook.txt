import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { chunkResources } from '@/hooks/shared/useChunk.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IChunkParams {
  size: number;
  filters?: Record<string, unknown>;
}

export const useChunk{{tableName.plural.pascalCase}} = (
  params: IChunkParams,
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[][]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'chunk', params.size, params.filters],
    queryFn: () =>
      chunkResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/chunk',
        params.size,
        params.filters
      ),
    ...options,
  });
};
