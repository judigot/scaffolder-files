import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { getWithRelationsResources } from '@/hooks/shared/useGetWithRelations.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IGetWithRelationsParams {
  relations: string[];
  filters?: Record<string, unknown>;
}

export const useGetWithRelations{{tableName.plural.pascalCase}} = (
  params: IGetWithRelationsParams,
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'withRelations', params.relations, params.filters],
    queryFn: () =>
      getWithRelationsResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/with-relations',
        params.relations,
        params.filters
      ),
    ...options,
  });
};
