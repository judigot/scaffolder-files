import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { firstOrCreateResource } from '@/hooks/shared/useFirstOrCreate.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IFirstOrCreateData {
  searchCriteria: Partial<I{{tableName.singular.pascalCase}}>;
  defaults?: Partial<I{{tableName.singular.pascalCase}}>;
}

export const useFirstOrCreate{{tableName.singular.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<I{{tableName.singular.pascalCase}}, Error, IFirstOrCreateData>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ searchCriteria, defaults }: IFirstOrCreateData) =>
      firstOrCreateResource<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/first-or-create',
        searchCriteria,
        defaults
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
    },
    ...options,
  });
};
