import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { whereBetweenResources } from '@/hooks/shared/useWhereBetween.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IWhereBetweenParams {
  column: string;
  min: string | number;
  max: string | number;
}

export const useWhereBetween{{tableName.plural.pascalCase}} = (
  params: IWhereBetweenParams,
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'whereBetween', params.column, params.min, params.max],
    queryFn: () =>
      whereBetweenResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/where-between',
        params.column,
        params.min,
        params.max
      ),
    ...options,
  });
};
