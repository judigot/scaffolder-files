import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { pluckResources } from '@/hooks/shared/usePluck.ts';

interface IPluckParams {
  column: string;
  key?: string;
}

export const usePluck{{tableName.plural.pascalCase}} = (
  params: IPluckParams,
  options?: Omit<
    UseQueryOptions<(string | number)[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'pluck', params.column, params.key],
    queryFn: () =>
      pluckResources(
        '/{{tableName.plural.camelCase}}/pluck',
        params.column,
        params.key
      ),
    ...options,
  });
};
