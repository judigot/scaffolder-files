import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
import { eachResource } from '@/hooks/shared/useEach.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface IEachParams {
  batchSize?: number;
  filters?: Record<string, unknown>;
  callback: (item: I{{tableName.singular.pascalCase}}) => void | Promise<void>;
}

export const useEach{{tableName.plural.pascalCase}} = (
  options?: Omit<
    UseMutationOptions<void, Error, IEachParams>,
    'mutationFn'
  >,
) => {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ batchSize, filters, callback }: IEachParams) =>
      eachResource<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/each',
        callback,
        batchSize,
        filters
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['{{tableName.plural.camelCase}}'] });
    },
    ...options,
  });
};
