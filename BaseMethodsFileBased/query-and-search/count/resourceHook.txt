import { useQuery, UseQueryOptions } from '@tanstack/react-query';
import { countResources } from '@/hooks/shared/useCount.ts';

interface ICountParams {
  filters?: Record<string, unknown>;
  distinct?: boolean;
  column?: string;
}

export const useCount{{tableNamePascalCasePlural}} = (
  params: ICountParams = {},
  options?: Omit<
    UseQueryOptions<number, Error>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableNameCamelCasePlural}}', 'count', params.filters, params.distinct, params.column],
    queryFn: () =>
      countResources(
        '/{{tableNameCamelCasePlural}}/count',
        params.filters,
        params.distinct,
        params.column
      ),
    ...options,
  });
};
