import { useQuery, type UseQueryOptions } from '@tanstack/react-query';
import { searchResources } from '@/hooks/shared/useSearch.ts';
import { type I{{tableName.singular.pascalCase}} } from '@/interfaces/I{{tableName.singular.pascalCase}}.ts';

interface ISearchParams {
  query: string;
  columns?: string[];
  filters?: Record<string, unknown>;
  limit?: number;
}

export const useSearch{{tableName.plural.pascalCase}} = (
  params: ISearchParams,
  options?: Omit<
    UseQueryOptions<I{{tableName.singular.pascalCase}}[]>,
    'queryKey' | 'queryFn'
  >,
) => {
  return useQuery({
    queryKey: ['{{tableName.plural.camelCase}}', 'search', params.query, params.columns, params.filters, params.limit],
    queryFn: () =>
      searchResources<I{{tableName.singular.pascalCase}}>(
        '/{{tableName.plural.camelCase}}/search',
        params.query,
        params.columns,
        params.filters,
        params.limit
      ),
    enabled: Boolean(params.query?.trim()),
    ...options,
  });
};
